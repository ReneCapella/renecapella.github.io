This circuit is an execise to learn how to use analogWrite, 
analogRead, digialWrite, and digitalRead
with 3 LEDs and one RGB LED

created 1 Nov 2008
by David A. Mellis
modified 19 Jan 2023

//original pin that feeds the circuit  
int ledPin = 6;
// the R in RGB diode set to off 
int red = 0;
// the G in RGB diode set to off 
int green = 0;
// the B in RGB diode set to off 
int blue = 0;

// the setup routine runs once when you press reset:
void setup() {
    //set pin 2 to INPUT 
    pinMode(2, INPUT);
    //set pin A0 to INPUT 
    pinMode(A0, INPUT);
    //set pin 4 to output
    pinMode(4, OUTPUT);
    //seed randomSeed with noise from pin 0
    randomSeed(analogRead(0));
}

// forever or until power is cut off..
void loop() {
    //increment in 5 integer values from 0 to 255...
    for (int fade = 0; fade <= 255; fade += 5) {

    //turn off that flickering white LED fed by pin 4
    digitalWrite(4, LOW);

    //set the blue LEDs fed by twoBlueLED to a value 
    between 0 and 250
    analogWrite(ledPin, fade);

    //if pin 2 detects power
    if (digitalRead(2) == 1) {
        //assign a random value to red
        red = random(255);
        //change the R in RGB to the red value
        analogWrite(9, red);
        //assign a random value to green
        green = random(255);
        //change the G in RGB to the green value
        analogWrite(10, green);
        //assign a random value to blue
        blue = random(255);
        //change the B in RGB to the blue value
        analogWrite(11, blue);
        //OTHERWISE... if pin 2 doesn't read 1
    } else {
        //turn the red off
        analogWrite(9, 0);
        //turn the green off
        analogWrite(10, 0);
        //turn the blue off
        analogWrite(11, 0);      
    }

    // wait for 80 milliseconds so we can see 
    how nice it is
    delay(80);
    }

    // fade out from max to min in increments 
    of 5 points:
    for (int fade = 255; fade >= 0; fade -= 5) {

    //while fading out..
    //turn off red
    analogWrite(9, 0);
    //turn off green
    analogWrite(10, 0);
    //turn off blue
    analogWrite(11, 0);

    //set the blue LEDs fed by twoBlueLED to a 
    value between 0 and 250
    analogWrite(ledPin, fade);

    //while we fade out, let's flicker a white LED
    //if button is pressed and input at pin 2 reads 1
    if (digitalRead(2) == 1) {
        // turn on pin 4
        digitalWrite(4, HIGH);
        //but if it *doesn;t* read 1...
    } else {
        //then turn that pin 4 LED off
        digitalWrite(4, LOW);
    }    

    // wait for 80 milliseconds so we can go slower
    delay(80);
    }
}

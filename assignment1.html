//set up once
void setup() {
    //initalization of pins 9, 10, & 11
    for (int pin = 9; i < 12; pin++){
        //sets pin to output
        pinMode(pin, OUTPUT);
    }
}

//loop forever
void loop() {
    //loop through pins 9, 10, & 11
    for (int pin = 9; pin < 12; pin++){
        //turn on the LED
        digitalWrite(pin, HIGH);
        //keep the LED on for one second
        delay(1000);
        //turn the LED off
        digitalWrite(pin, LOW);
        //keep the LED off for one second
        delay(1000);
    }
}
